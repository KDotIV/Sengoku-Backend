query EventEntrants($perPage: Int!, $eventSlug: String!) {
  event(slug: $eventSlug) {
    id
    name
    entrants(query: {perPage: $perPage, filter: {}}) {
      nodes {
        id
        participants {
          id
          gamerTag
        }
        standing {
          id
          placement
          container {
            __typename
            ... on Tournament {
              id
              name
              countryCode
              startAt
              endAt
              events {
                id
                name
              }
            }
            ... on Event {
              id
              name
              startAt
              numEntrants
              tournament {
                id
                name
              }
            }
            ... on Set {
              id
              event {
                id
                name
              }
              startAt
              completedAt
              games {
                id
              }
            }
          }
        }
      }
    }
  }
}
